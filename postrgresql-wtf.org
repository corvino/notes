#+BEGIN_SRC conf
local   all             flyway                                  password
#+END_SRC

*** [[https://www.postgresql.org/docs/9.6/static/auth-pg-hba-conf.html][The pg_hba.conf File]] docs
** Create a db, login role (user), and grant permissions
#+BEGIN_SRC SQL
create database flyway;
create role flyway with login password 'flyway';
grant all privileges on database flyway to flyway;

drop database flyway;
drop role flyway;
#+END_SRC
