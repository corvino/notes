* Using a pointer to a repo
Examples

#+BEGIN_SRC shell
  git init --separate-git-dir=SEP_DIR .

  git clone --separate-git-dir=../repo git@github.com:corvino/notes.git .
  # no check out
  git clone -n --separate-git-dir=../repo git@github.com:corvino/notes.git .
  git clone --separate-git-dir=repo git@github.com:corvino/notes.git wt

  # Add worktree, for a separate checkout
  git worktree add ../new HEAD
  git worktree list

  # Clone a new repo to a specified location, with separate directory
  # point to location, but no files checkout out. The separate directory
  # must be empty, but files can be copied to it after to link existing
  # files to a new clone.
  git clone -n --separate-git-dir=new-repo git@github.com:corvino/notes.git new-wt
  git restore --staged .
  cp ...
#+END_SRC
